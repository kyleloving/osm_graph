1use lru::LruCache;
2use std::sync::Mutex;

5use crate::graph::SpatialGraph;

8type GraphCache = Mutex<LruCache<String, SpatialGraph>>;
9type XmlCache = Mutex<LruCache<String, String>>;
10
11lazy_static! {
12    static ref GRAPH_CACHE: GraphCache =
13        Mutex::new(LruCache::new(std::num::NonZeroUsize::new(100).unwrap()));
14    static ref XML_CACHE: XmlCache =
15        Mutex::new(LruCache::new(std::num::NonZeroUsize::new(20).unwrap()));
16}
17
18pub fn check_cache(key: &str) -> Result<Option<SpatialGraph>, OsmGraphError> {
19    Ok(GRAPH_CACHE.lock().map_err(|_| OsmGraphError::LockPoisoned)?.get(key).cloned())
20}
21
22pub fn insert_into_cache(key: String, sg: SpatialGraph) -> Result<(), OsmGraphError> {
23    GRAPH_CACHE.lock().map_err(|_| OsmGraphError::LockPoisoned)?.put(key, sg);
24    Ok(())
25}
26
27pub fn check_xml_cache(query: &str) -> Result<Option<String>, OsmGraphError> {
28    Ok(XML_CACHE.lock().map_err(|_| OsmGraphError::LockPoisoned)?.get(query).cloned())
29}
30
31pub fn insert_into_xml_cache(query: String, xml: String) -> Result<(), OsmGraphError> {
32    XML_CACHE.lock().map_err(|_| OsmGraphError::LockPoisoned)?.put(query, xml);
33    Ok(())
34}
35
36pub fn clear_cache() -> Result<(), OsmGraphError> {
37    GRAPH_CACHE.lock().map_err(|_| OsmGraphError::LockPoisoned)?.clear();
38    XML_CACHE.lock().map_err(|_| OsmGraphError::LockPoisoned)?.clear();
39    Ok(())
40}

44/// FNV-1a 64-bit hash — stable across Rust versions, no dependencies.
45fn fnv1a(s: &str) -> u64 {
46    let mut hash: u64 = 14695981039346656037;
47    for byte in s.bytes() {
48        hash ^= byte as u64;
49        hash = hash.wrapping_mul(1099511628211);
50    }
51    hash
52}

54/// Returns the disk cache directory, overridable via `OSM_GRAPH_CACHE_DIR`.
55/// Defaults to a `cache/` folder in the current working directory — the same
56/// convention used by OSMnx, so researchers get persistent, visible caching
57/// next to their notebooks and scripts.
58pub fn disk_cache_dir() -> PathBuf {
59    std::env::var_os("OSM_GRAPH_CACHE_DIR")
60        .map(PathBuf::from)
61        .unwrap_or_else(|| PathBuf::from("cache"))
62}
63
64fn disk_xml_path(query: &str) -> PathBuf {
65    disk_cache_dir().join(format!("{:016x}.xml", fnv1a(query)))
66}
67
68/// Check the disk cache for a previously fetched Overpass XML response.
69/// Returns None on any error — a cache miss is always safe.
70pub fn check_disk_xml_cache(query: &str) -> Option<String> {
71    std::fs::read_to_string(disk_xml_path(query)).ok()
72}
73
74/// Persist an Overpass XML response to disk. Best-effort — silently ignores errors.
75pub fn write_disk_xml_cache(query: &str, xml: &str) {
76    let dir = disk_cache_dir();
77    if std::fs::create_dir_all(&dir).is_ok() {
78        let _ = std::fs::write(disk_xml_path(query), xml);
79    }
80}
81
82/// Delete all files in the disk cache directory.
83pub fn clear_disk_cache() -> Result<(), OsmGraphError> {
84    let dir = disk_cache_dir();
85    if dir.exists() {
86        std::fs::remove_dir_all(&dir)?;
87    }
88    Ok(())
89}

95    #[test]
96    fn test_disk_cache_round_trip() {
97        // Use a nanosecond-suffixed temp dir to avoid collisions with parallel test runs
98        let dir = std::env::temp_dir().join(format!(
99            "osm_graph_test_{}",
100            std::time::SystemTime::now()
101                .duration_since(std::time::UNIX_EPOCH)
102                .unwrap()
103                .subsec_nanos()
104        ));
105        std::env::set_var("OSM_GRAPH_CACHE_DIR", &dir);
106
107        write_disk_xml_cache("test_query", "<xml>hello</xml>");
108        let result = check_disk_xml_cache("test_query");
109        assert_eq!(result, Some("<xml>hello</xml>".to_string()));
110
111        assert!(check_disk_xml_cache("other_query").is_none());
112
113        let _ = std::fs::remove_dir_all(&dir);
114        std::env::remove_var("OSM_GRAPH_CACHE_DIR");
115    }
